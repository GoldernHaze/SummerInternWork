## 1. The Big Picture — Why This Pipeline Exists

Mentor Lucas needs a system that takes data from a satellite payload instrument (triggered by an "occultation" event) and transmits it to Earth via radio. Problem: this data can be large, but radio packets (NGHam protocol) can only carry a limited number of bytes per packet (max 220 bytes).

**Solution — a 3-stage pipeline**, where each stage is an independent FreeRTOS task:

```
[Payload Memory] -> TASK 1 -> TASK 2 -> TASK 3 -> [Radio Transmission]
                 (fragment)  (FIFO queue)  (NGHam wrap)
```

- **Task 1 (this document):** Read payload data (simulated for now), split it into fixed-size chunks, attach metadata (occultation ID, sequence number, total chunks, data size) to each chunk, and push each chunk into a queue.
- **Task 2 (not yet built):** Receive chunks from Task 1's queue, preserve strict ordering (FIFO), forward to a second queue.
- **Task 3 (not yet built):** Receive chunks from Task 2's queue, wrap each one in a full NGHam packet, and "transmit" it (simulated for now via print/log).

Lucas's explicit recommendation: build this first with **simulated memory** (no real hardware), then port to the real MSP430F6659 microcontroller later using Code Composer Studio.

## 3. Full `main.c` Source Code (as built and tested)

```c
#include <stdio.h>
#include <string.h>
#include "FreeRTOS.h"
#include "task.h"
#include "queue.h"
#include "payload_chunk.h"

/* Queue between Task 1 (producer) and the temporary consumer */
QueueHandle_t xChunkQueue;

/* ---------- Simulated occultation "events" ---------- */
typedef struct {
    uint16_t occultation_id;
    uint16_t data_size;   /* how many bytes of fake payload this event has */
} FakeEvent_t;

static const FakeEvent_t fake_events[] = {
    { .occultation_id = 1000, .data_size = 100 },
    { .occultation_id = 1001, .data_size = 45  },
    { .occultation_id = 1002, .data_size = 70  },
};

#define NUM_FAKE_EVENTS (sizeof(fake_events) / sizeof(fake_events[0]))

/* ---------- TASK 1: Payload Reader / Fragmenter ---------- */
void vTask1_PayloadReader(void *pvParameters)
{
    (void) pvParameters;

    for (int e = 0; e < (int)NUM_FAKE_EVENTS; e++)
    {
        uint16_t occ_id   = fake_events[e].occultation_id;
        uint16_t data_len = fake_events[e].data_size;

        /* Simulate fake payload bytes (just an incrementing pattern) */
        uint8_t fake_payload[256];
        for (int i = 0; i < data_len; i++)
        {
            fake_payload[i] = (uint8_t)(0x10 + (e * 0x30) + i);
        }

        /* Calculate how many chunks this event needs */
        uint8_t total_chunks = (data_len + CHUNK_DATA_SIZE - 1) / CHUNK_DATA_SIZE;

        printf("[Task1-PayloadReader] occultation %d: %d bytes -> %d chunk(s)\n",
               occ_id, data_len, total_chunks);
        fflush(stdout);

        int bytes_remaining = data_len;
        int offset = 0;

        for (uint8_t seq = 1; seq <= total_chunks; seq++)
        {
            PayloadChunk_t chunk;
            memset(&chunk, 0, sizeof(chunk));

            uint8_t this_chunk_size = (bytes_remaining < CHUNK_DATA_SIZE)
                                        ? (uint8_t)bytes_remaining
                                        : CHUNK_DATA_SIZE;

            chunk.occultation_id  = occ_id;
            chunk.sequence_number = seq;
            chunk.total_chunks    = total_chunks;
            chunk.data_size       = this_chunk_size;
            memcpy(chunk.data, &fake_payload[offset], this_chunk_size);

            /* Send this chunk into the queue */
            xQueueSend(xChunkQueue, &chunk, portMAX_DELAY);

            offset          += this_chunk_size;
            bytes_remaining -= this_chunk_size;

            vTaskDelay(pdMS_TO_TICKS(50)); /* small delay, simulate real work */
        }
    }

    printf("[Task1-PayloadReader] all occultations fragmented. Task finished.\n");
    fflush(stdout);

    /* Task 1 is done - just idle forever (don't delete, keep it simple) */
    for (;;)
    {
        vTaskDelay(pdMS_TO_TICKS(1000));
    }
}

/* ---------- TEMPORARY dummy consumer (stand-in for Task 2/3) ---------- */
void vTempConsumer(void *pvParameters)
{
    (void) pvParameters;
    PayloadChunk_t received;
    int total_received = 0;

    printf("[TempConsumer] started (placeholder - will become Task2/FIFO next).\n");
    fflush(stdout);

    for (;;)
    {
        if (xQueueReceive(xChunkQueue, &received, pdMS_TO_TICKS(3000)) == pdTRUE)
        {
            total_received++;
            printf("[TempConsumer] got chunk  occ=%d  seq=%d/%d  bytes=%d  first_byte=0x%02X\n",
                   received.occultation_id,
                   received.sequence_number,
                   received.total_chunks,
                   received.data_size,
                   received.data[0]);
            fflush(stdout);
        }
        else
        {
            printf("[TempConsumer] no more chunks arriving. Total chunks received: %d\n",
                   total_received);
            fflush(stdout);
            for (;;) vTaskDelay(pdMS_TO_TICKS(1000)); /* idle forever */
        }
    }
}

int main(void)
{
    printf("=== CubeSat Packet Pipeline - Task 1 Test ===\n\n");
    fflush(stdout);

    xChunkQueue = xQueueCreate(10, sizeof(PayloadChunk_t));

    xTaskCreate(vTask1_PayloadReader, "Task1", configMINIMAL_STACK_SIZE * 2, NULL, 1, NULL);
    xTaskCreate(vTempConsumer, "TempConsumer", configMINIMAL_STACK_SIZE * 2, NULL, 1, NULL);

    vTaskStartScheduler();

    for (;;);
    return 0;
}

/* Required by FreeRTOSConfig.h for configASSERT */
void vAssertCalled(const char *pcFile, unsigned long ulLine)
{
    printf("ASSERT FAILED: %s, line %lu\n", pcFile, ulLine);
    fflush(stdout);
    for (;;);
}

/* Required because configUSE_MALLOC_FAILED_HOOK is 1 */
void vApplicationMallocFailedHook(void)
{
    printf("MALLOC FAILED! Out of heap memory.\n");
    fflush(stdout);
    for (;;);
}

/* Required because configCHECK_FOR_STACK_OVERFLOW is 2 */
void vApplicationStackOverflowHook(TaskHandle_t xTask, char *pcTaskName)
{
    (void) xTask;
    printf("STACK OVERFLOW in task: %s\n", pcTaskName);
    fflush(stdout);
    for (;;);
}
```

### 4.1 Includes

```c
#include <stdio.h>          // printf()

#include "FreeRTOS.h"        // core FreeRTOS types/macros

```

Standard C libraries plus the FreeRTOS headers we need, plus our own custom struct definition.

### 4.2 The Queue Handle

```c
QueueHandle_t xChunkQueue;
```

Declared globally (outside any function) so that **both** Task 1 and `vTempConsumer` can access the same "mailbox." This is the communication channel between them.

### 4.3 Fake Event Data (simulating payload memory)

```c
typedef struct {
    uint16_t occultation_id;
    uint16_t data_size;
} FakeEvent_t;

static const FakeEvent_t fake_events[] = {
    { .occultation_id = 1000, .data_size = 100 },
    { .occultation_id = 1001, .data_size = 45  },
    { .occultation_id = 1002, .data_size = 70  },
};
```

Since there's no real camera/sensor generating real occultation data yet, we invented 3 fake events with different sizes — one exact multiple of the chunk size (100), one smaller than a full chunk on the last piece (45), one in between (70) — specifically to test that the fragmentation math handles all cases correctly (exact fits, small remainders, larger remainders).

### 4.4 Task 1 — Outer Loop (looping through each fake event)

```c
for (int e = 0; e < (int)NUM_FAKE_EVENTS; e++)
{
    uint16_t occ_id   = fake_events[e].occultation_id;
    uint16_t data_len = fake_events[e].data_size;
```

Walks through the 3 fake events one at a time.

### 4.5 Generating Fake Payload Bytes

```c
uint8_t fake_payload[256];
for (int i = 0; i < data_len; i++)
{
    fake_payload[i] = (uint8_t)(0x10 + (e * 0x30) + i);
}
```

Generates fake bytes with a predictable, incrementing pattern, shifted per event using `e * 0x30`. This is purely a debugging trick — it lets us visually confirm "yes, this chunk really does contain the data I expect, in the right order" by checking the printed `first_byte` value.

**Verified against real output:**
- Event 0 (occ 1000): bytes start at `0x10` -> first chunk shows `first_byte=0x10`
- Event 1 (occ 1001): bytes start at `0x10 + 1*0x30 = 0x40` -> shows `first_byte=0x40`
- Event 2 (occ 1002): bytes start at `0x10 + 2*0x30 = 0x70` -> shows `first_byte=0x70`

### 4.6 Calculating Total Chunks Needed (ceiling division)

```c
uint8_t total_chunks = (data_len + CHUNK_DATA_SIZE - 1) / CHUNK_DATA_SIZE;
```

Normal integer division in C always rounds DOWN (e.g. `100 / 32 = 3`, discarding the remainder), but we need to round UP, since even 1 leftover byte still needs its own chunk. Adding `CHUNK_DATA_SIZE - 1` (i.e. 31) before dividing forces correct rounding up. This is a standard C idiom called "ceiling division."

**Verified with real numbers:**

| data_len | Calculation | Result | Matches output? |
|---|---|---|---|
| 100 | (100+31)/32 = 131/32 = 4 (integer division truncates) | 4 chunks | Yes |
| 45  | (45+31)/32 = 76/32 = 2  | 2 chunks | Yes |
| 70  | (70+31)/32 = 101/32 = 3 | 3 chunks | Yes |

### 4.7 Inner Loop — Building Each Individual Chunk

```c
int bytes_remaining = data_len;
int offset = 0;

for (uint8_t seq = 1; seq <= total_chunks; seq++)
{
    PayloadChunk_t chunk;
    memset(&chunk, 0, sizeof(chunk));

    uint8_t this_chunk_size = (bytes_remaining < CHUNK_DATA_SIZE)
                                ? (uint8_t)bytes_remaining
                                : CHUNK_DATA_SIZE;
```

Traced exactly for occultation 1000 (100 bytes):

| Iteration (seq) | bytes_remaining (before) | this_chunk_size | bytes_remaining (after) |
|---|---|---|---|
| 1 | 100 | min(100,32) = 32 | 68 |
| 2 | 68  | min(68,32) = 32  | 36 |
| 3 | 36  | min(36,32) = 32  | 4  |
| 4 | 4   | min(4,32) = 4    | 0  |

This exactly matches the real output: `seq=1/4 bytes=32`, `seq=2/4 bytes=32`, `seq=3/4 bytes=32`, `seq=4/4 bytes=4`. The last chunk is correctly smaller because only 4 bytes remained.

### 4.8 Filling In the PayloadChunk_t "Form"

```c
chunk.occultation_id  = occ_id;
chunk.sequence_number = seq;
chunk.total_chunks    = total_chunks;
chunk.data_size       = this_chunk_size;
memcpy(chunk.data, &fake_payload[offset], this_chunk_size);
```

This fills out the 5 fields of the `PayloadChunk_t` struct (see Section 5.1). `memcpy` copies `this_chunk_size` bytes from the correct position (`offset`) in `fake_payload` into the chunk's `data` array.

### 4.9 Sending the Chunk Into the Queue

```c
xQueueSend(xChunkQueue, &chunk, portMAX_DELAY);

offset          += this_chunk_size;
bytes_remaining -= this_chunk_size;

vTaskDelay(pdMS_TO_TICKS(50));
```

`xQueueSend` copies the entire filled-out `chunk` struct into the queue's internal storage. `offset` and `bytes_remaining` are updated for the next iteration. The small `vTaskDelay(50)` simulates realistic work time, and also lets FreeRTOS switch over to `vTempConsumer` so it can actually drain the queue as chunks arrive, rather than Task 1 hogging the CPU.

### 4.10 The Temporary Consumer — `vTempConsumer`

```c
if (xQueueReceive(xChunkQueue, &received, pdMS_TO_TICKS(3000)) == pdTRUE)
{
    total_received++;
    printf("[TempConsumer] got chunk  occ=%d  seq=%d/%d  bytes=%d  first_byte=0x%02X\n", ...);
}
else
{
    printf("[TempConsumer] no more chunks arriving. Total chunks received: %d\n", total_received);
    ...
}
```

Waits up to 3 seconds for a new chunk. If one arrives, prints its details and counts it. If 3 full seconds pass with nothing arriving (meaning Task 1 must be finished), prints the final total and goes idle. This stands in for the real Task 2/Task 3 for now, purely to verify Task 1's correctness in isolation.

### 4.11 `main()` — Wiring Everything Together

```c
xChunkQueue = xQueueCreate(10, sizeof(PayloadChunk_t));

xTaskCreate(vTask1_PayloadReader, "Task1", configMINIMAL_STACK_SIZE * 2, NULL, 1, NULL);
xTaskCreate(vTempConsumer, "TempConsumer", configMINIMAL_STACK_SIZE * 2, NULL, 1, NULL);

vTaskStartScheduler();
```

1. Creates the queue (mailbox) that can hold up to 10 `PayloadChunk_t` items at once.
2. Registers both tasks with FreeRTOS.
3. `vTaskStartScheduler()` hands control to FreeRTOS permanently — from here on, FreeRTOS automatically switches between the two tasks whenever one calls a delay or a blocking queue operation.

### 4.12 The Hook Functions

```c
void vAssertCalled(...) { ... }
void vApplicationMallocFailedHook(void) { ... }
void vApplicationStackOverflowHook(...) { ... }
```

Safety nets required because `FreeRTOSConfig.h` enables `configASSERT`, `configUSE_MALLOC_FAILED_HOOK`, and `configCHECK_FOR_STACK_OVERFLOW`. They never actually triggered in our test run (no errors occurred), but the program would not compile/link without them being defined somewhere.

#ifndef PAYLOAD_CHUNK_H

#include <stdint.h>

#define CHUNK_DATA_SIZE 32   /* placeholder size, matches Amrit's approach */

typedef struct {
    uint16_t occultation_id;     /* which event generated this data */

This defines the exact "envelope" shape every chunk uses. Every chunk this pipeline creates is one filled-out instance of this struct, and it travels as a single unit through the queues.

### 5.3 `Makefile`

```makefile
# ---- Paths ----
KERNEL_DIR = FreeRTOS-Kernel
PORT_DIR = $(KERNEL_DIR)/portable/ThirdParty/GCC/Posix
UTILS_DIR = $(PORT_DIR)/utils
MEMMANG_DIR = $(KERNEL_DIR)/portable/MemMang

# ---- Compiler ----
CC = gcc
CFLAGS = -Wall -Wextra -O0 -g \
	-I src \
	-I $(KERNEL_DIR)/include \
	-I $(PORT_DIR) \
	-I $(UTILS_DIR) \
	-D_POSIX_C_SOURCE=200809L

LDFLAGS = -lpthread

# ---- Source files ----
SRCS = src/main.c \
	$(KERNEL_DIR)/tasks.c \
	$(KERNEL_DIR)/queue.c \
	$(KERNEL_DIR)/list.c \
	$(KERNEL_DIR)/timers.c \
	$(KERNEL_DIR)/event_groups.c \
	$(MEMMANG_DIR)/heap_3.c \
	$(PORT_DIR)/port.c \
	$(UTILS_DIR)/wait_for_event.c

OBJS = $(SRCS:.c=.o)

TARGET = build/cubesat_pipeline

# ---- Rules ----
all: $(TARGET)

$(TARGET): $(OBJS)
	mkdir -p build
	$(CC) $(OBJS) -o $(TARGET) $(LDFLAGS)

%.o: %.c
	$(CC) $(CFLAGS) -c $< -o $@

clean:
	rm -f $(OBJS) $(TARGET)

run: all
	./$(TARGET)

.PHONY: all clean run
```

Note: `LDFLAGS` originally included `-lrt` (needed on Linux for POSIX real-time timer functions), but this was removed for macOS, since that functionality is already built into macOS's core system libraries and no separate `librt` exists there.

**macOS-specific fix also applied to `port.c`:** the line calling `pthread_setname_np(pxThreadName)` was commented out, since macOS's version of this function has a different signature than Linux's under the specific `_POSIX_C_SOURCE` flag used here. This only affects cosmetic debug thread-naming, not functionality.

## 7. What Is Real vs. Simulated Right Now

| Part | Real or Simulated? |
|---|---|
| FreeRTOS tasks/scheduling | Real — same code will run unchanged on the real MSP430 later |
| Queue behavior (`xQueueSend`/`xQueueReceive`) | Real — same code will run unchanged later |
| Payload data (`fake_payload[]`) | Simulated — invented bytes; later replaced with real payload memory reads |
| The 3 occultation events | Simulated — invented for testing; later triggered by a real instrument event |
| The microcontroller (Mac instead of MSP430) | Simulated — using FreeRTOS's POSIX port; later needs recompiling for MSP430 target |
| NGHam wrapping | Not built yet — this is Task 3, still ahead |

**Open question for Lucas (raised by teammate Amrit, still pending):** is 32 bytes (`CHUNK_DATA_SIZE`) the correct chunk size?
